class Student{
public:
    string name;

        surname = s;
        lecture_hall = m;

    };
};
class Teacher: public Student{
public:
    list<Student*> group;
    short rank;

        name = n1;
        surname = s1;
        lecture_hall = m1;

    };
    Teacher& visit_card(){
        cout << "Surname: " << surname << endl;
        cout << "Name: " << name << endl;
        cout << "Lecture hall: " << lecture_hall << endl;

    }
};
